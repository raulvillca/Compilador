include macros2.asm
include number.asm

.MODEL	LARGE
.386
.STACK 200h

MAXTEXTSIZE equ 50

.DATA

.CODE

START:

CONDWHILE:

MOV R1, _ID
CMP R1, 2				;WHILE ID<2
JGE FWHILE
					;VERDADERO
MOV @AUX, _ID3
MOV DX, OFFSET @AUX			
MOV AH, 09H				;Print
INT 21H

FWHILE:				;FALSO WHILE

MOV R1, _ID1
CMP R1, 23				;IF ID<23
JGE FIF1
					;VERDADERO
MOV R1, _ID2
MOV _ID2, 2

FIF1:				;FALSO IF

MOV R1, _ID3
CMP R1, 2				;IF ID3==2
JNE FIF2
					;VERDADERO
MOV R1 _ID4
CMP 3, _ID4				;IF 3<=ID4
JG	FIF2 
					;VERDADERO
MOV @AUX, WRITE 
ADD @AUX, _ID3

MOV @AUX2, "Hola Mundo"
MOV DX, OFFSET @AUX2
MOV AH, 09H
INT 21H					;Print
MOV AH,4CH
INT 21H

END START
